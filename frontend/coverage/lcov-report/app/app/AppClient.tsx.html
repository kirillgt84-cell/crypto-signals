function formatTfLabel(tf: string) {
  const mapping: Record<string, string> = {
    "15": "15m",
    "60": "1h",
    "240": "4h",
    "D": "1d",
    "3D": "3d",
    "1W": "1w",
  }
  return mapping[tf] || tf
}
